例题9.1玛雅人的密码（清华大学上机）
利用一个mp来标记是否出现过，之后进行BFS，且每一次shift之后mp都将加1

#include<iostream>
#include<cstdio>
#include<queue>
#include<string>
#include<vector>
#include<map> 
using namespace std;
map<string, int> mp;//记录每一个字符的交换次数 
queue<string> que;//用于bfs的队列 
int n;
string s;
bool has_2012(string s){ //判断是否含有2012 
    for(int i = 0; i < n - 3; i++){
        if(s[i] == '2' && s[i+1] == '0' && s[i+2] == '1' && s[i+3] == '2')
            return true;
    }
    return false;
}
string shift(string s, int i){//用来转换字符串 , i位与i+1交换 
        char temp = s[i];
        s[i] = s[i+1];
        s[i+1] = temp;
        return s;
}
int bfs(string s){
    string newstr;
    mp.clear();
    while(!que.empty()) que.pop();//清空队列，队列没有clear（）函数 
    que.push(s);//初始字符串入队 
    mp[s] = 0;//初始字符串交换次数为0 
    while(!que.empty()){
        s = que.front();
        que.pop();//取出队首 
        for(int i = 0; i < n - 1; i++){
            newstr = shift(s, i);
            if(mp.find(newstr) == mp.end()){//字符没有出现过 
                mp[newstr] = mp[s] + 1;//出现了且交换次数多于他爹的交换次数 
                if(has_2012(newstr)) return mp[newstr];
                else que.push(newstr);
            }
        }
    }
    return -1;
}

int main(){
    while(cin >> n >> s){
        if(has_2012(s)) cout << 0 << endl;
        else cout << bfs(s) << endl;
    }
    return 0;
}

        if(n == 1 || n == 0 || n < 0){
            flag =1;
        }
